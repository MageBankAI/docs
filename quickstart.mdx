This guide will help you get started with the MageBank API quickly. You'll learn how to obtain your API key, create your first agent, and execute a basic transaction.

## Prerequisites

Before you begin, make sure you have:

1. Created a MageBank account
2. Some USDC in your wallet (for mainnet) or test USDC (for testnet)

## Step 1: Get Your API Key

<Steps>
  <Step title="Log in to MageBank">
    Navigate to the [MageBank Dashboard](https://www.magebank.ai/dashboard)
  </Step>
  <Step title="Navigate to API Settings">
    In the sidebar, click on "Integrate Agent"
  </Step>
  <Step title="Copy your API Key">
    Your API key should be displayed in the "API Key" section. Click "Copy".
    
    <Warning>
      Keep your API key secure. Anyone with your key can perform actions on behalf of your account.
    </Warning>
  </Step>
</Steps>

## Step 2: Make Your First API Request

Let's make a simple API request to verify your API key works. We'll check your wallet balance:

<CodeGroup>

```bash cURL
curl -X GET "https://api.magebank.ai/user/wallet-balance" \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript Node.js
const axios = require('axios');

async function getWalletBalance() {
  try {
    const response = await axios.get('https://api.magebank.ai/user/wallet-balance', {
      headers: {
        'x-api-key': 'YOUR_API_KEY'
      }
    });
    console.log(response.data);
  } catch (error) {
    console.error('Error:', error.response ? error.response.data : error.message);
  }
}

getWalletBalance();
```

```python Python
import requests

def get_wallet_balance():
    url = "https://api.magebank.ai/user/wallet-balance"
    headers = {
        "x-api-key": "YOUR_API_KEY"
    }
    
    response = requests.get(url, headers=headers)
    
    if response.status_code == 200:
        print(response.json())
    else:
        print(f"Error: {response.status_code}")
        print(response.text)

get_wallet_balance()
```

</CodeGroup>

If successful, you should receive a response similar to:

```json
{
  "success": true,
  "balance": "100.50",
  "asset": "USDC",
  "message": "Balance retrieved successfully",
  "source": "blockchain"
}
```

## Step 3: Create Your First Agent

Let's create an AI agent that can handle financial transactions:

<CodeGroup>

```bash cURL
curl -X POST "https://api.magebank.ai/agents/create" \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "userid": "your_user_id",
    "name": "Payment Assistant",
    "description": "Handles payment processing for customer support",
    "balance": 50,
    "dailylimit": 1000,
    "transactionlimit": 100,
    "requireapprovalforall": false,
    "requireapprovalaboveamount": 50,
    "tags": ["customer-support", "payments"]
  }'
```

```javascript Node.js
const axios = require('axios');

async function createAgent() {
  try {
    const response = await axios.post('https://api.magebank.ai/agents/create', {
      userid: "your_user_id",
      name: "Payment Assistant",
      description: "Handles payment processing for customer support",
      balance: 50,
      dailylimit: 1000,
      transactionlimit: 100,
      requireapprovalforall: false,
      requireapprovalaboveamount: 50,
      tags: ["customer-support", "payments"]
    }, {
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    });
    
    console.log('Agent created:', response.data);
    // Store the agent ID for future use
    const agentId = response.data.id;
    console.log('Agent ID:', agentId);
  } catch (error) {
    console.error('Error:', error.response ? error.response.data : error.message);
  }
}

createAgent();
```

```python Python
import requests
import json

def create_agent():
    url = "https://api.magebank.ai/agents/create"
    headers = {
        "x-api-key": "YOUR_API_KEY",
        "Content-Type": "application/json"
    }
    
    payload = {
        "userid": "your_user_id",
        "name": "Payment Assistant",
        "description": "Handles payment processing for customer support",
        "balance": 50,
        "dailylimit": 1000,
        "transactionlimit": 100,
        "requireapprovalforall": False,
        "requireapprovalaboveamount": 50,
        "tags": ["customer-support", "payments"]
    }
    
    response = requests.post(url, headers=headers, data=json.dumps(payload))
    
    if response.status_code == 200:
        data = response.json()
        print("Agent created:", data)
        # Store the agent ID for future use
        agent_id = data["id"]
        print("Agent ID:", agent_id)
    else:
        print(f"Error: {response.status_code}")
        print(response.text)

create_agent()
```

</CodeGroup>

Upon success, you'll receive a response containing details about your new agent, including its ID. Make note of this ID as you'll need it for future API calls.

## Step 4: Make a Payment

Now that you have an agent, let's make a payment:

<CodeGroup>

```bash cURL
curl -X POST "https://api.magebank.ai/payments/register" \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "senderagentid": "YOUR_AGENT_ID",
    "receiveragentid": "RECIPIENT_AGENT_ID",
    "paymentdetails": {
      "amount": 10,
      "currency": "USDC",
      "method": "CRYPTO_ADDRESS"
    },
    "name": "Test Payment",
    "type": "EXTERNAL"
  }'
```

```javascript Node.js
const axios = require('axios');

async function makePayment() {
  try {
    const response = await axios.post('https://api.magebank.ai/payments/register', {
      senderagentid: "YOUR_AGENT_ID",
      receiveragentid: "RECIPIENT_AGENT_ID",
      paymentdetails: {
        amount: 10,
        currency: "USDC",
        method: "CRYPTO_ADDRESS"
      },
      name: "Test Payment",
      type: "EXTERNAL"
    }, {
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    });
    
    console.log('Payment created:', response.data);
    const paymentId = response.data.id;
    console.log('Payment ID:', paymentId);
  } catch (error) {
    console.error('Error:', error.response ? error.response.data : error.message);
  }
}

makePayment();
```

If the payment requires approval (based on your agent's rules), you'll need to approve it:

<CodeGroup>

```bash cURL
curl -X POST "https://api.magebank.ai/payments/setApprove" \
  -H "x-api-key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "paymentId": "PAYMENT_ID"
  }'
```

```javascript Node.js
const axios = require('axios');

async function approvePayment(paymentId) {
  try {
    const response = await axios.post('https://api.magebank.ai/payments/setApprove', {
      paymentId: paymentId
    }, {
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      }
    });
    
    console.log('Payment approved:', response.data);
  } catch (error) {
    console.error('Error:', error.response ? error.response.data : error.message);
  }
}

approvePayment("PAYMENT_ID");
```

```python Python
import requests
import json

def approve_payment(payment_id):
    url = "https://api.magebank.ai/payments/setApprove"
    headers = {
        "x-api-key": "YOUR_API_KEY",
        "Content-Type": "application/json"
    }
    
    payload = {
        "paymentId": payment_id
    }
    
    response = requests.post(url, headers=headers, data=json.dumps(payload))
    
    if response.status_code == 200:
        print("Payment approved:", response.json())
    else:
        print(f"Error: {response.status_code}")
        print(response.text)

approve_payment("PAYMENT_ID")
```

</CodeGroup>
